-- ══════════════════════════════════════════════════════════════════════════ -- AXC Planificación · Caja Chica v7.16 · RLS gestión (supervisora de caja chica) -- Permite que el SUPERVISOR de caja (o un usuario con el permiso puntual) -- ESCRIBA en TODAS las cajas: cambiar topes, acreditar, aprobar comprobantes, -- aprobar excepciones y aprobar/cerrar rendiciones. -- -- Proyecto Supabase: zvucnezwhgfgupasgqnd (Conexdatapy) -- -- Contexto: la app ya habilita estas acciones por permiso (cc_configurar, -- cc_acreditar, cc_aprobar, cc_aprobar_rendicion). Pero la BASE (RLS) solo deja -- escribir en la caja propia. Dahiana (supervisora) no podía acreditar/cambiar -- topes en cajas ajenas. Este script lo destraba a nivel base. -- -- SEGURO y ADITIVO: -- • NO borra ni cambia las políticas existentes (la de "caja propia" sigue). -- • NO habilita/deshabilita RLS. -- • Solo AGREGA políticas de lectura/escritura para gestores (se combinan con OR). -- • Un residente común (sin esos permisos) NO gana acceso: las funciones -- devuelven false para él y sigue con su política de caja propia. -- • NO otorga DELETE (borrado) a nadie nuevo, por seguridad. -- • Idempotente: se puede correr varias veces. -- ══════════════════════════════════════════════════════════════════════════ -- ── 1) Funciones helper (SECURITY DEFINER para leer 'perfiles' sin recursión) ── -- ¿El usuario puede VER todas las cajas? (admin / supervisor_caja / permiso cc_ver_todas) create or replace function public.cc_puede_ver_todas() returns boolean language sql stable security definer set search_path = public as $$ select exists ( select 1 from public.perfiles p where p.id = auth.uid() and (p.rol in ('admin','supervisor_caja') or (p.permisos ->> 'cc_ver_todas') = 'true') ); $$; grant execute on function public.cc_puede_ver_todas() to authenticated; -- ¿Tiene un permiso puntual? (o es admin/supervisor_caja) create or replace function public.cc_perm(perm text) returns boolean language sql stable security definer set search_path = public as $$ select exists ( select 1 from public.perfiles p where p.id = auth.uid() and (p.rol in ('admin','supervisor_caja') or (p.permisos ->> perm) = 'true') ); $$; grant execute on function public.cc_perm(text) to authenticated; -- ¿Es "gestor" de caja chica? (admin/supervisor o cualquier permiso de gestión) — para el historial create or replace function public.cc_es_gestor() returns boolean language sql stable security definer set search_path = public as $$ select exists ( select 1 from public.perfiles p where p.id = auth.uid() and ( p.rol in ('admin','supervisor_caja') or (p.permisos ->> 'cc_configurar') = 'true' or (p.permisos ->> 'cc_acreditar') = 'true' or (p.permisos ->> 'cc_aprobar') = 'true' or (p.permisos ->> 'cc_aprobar_rendicion') = 'true' or (p.permisos ->> 'cc_excepcion') = 'true' ) ); $$; grant execute on function public.cc_es_gestor() to authenticated; -- ── 2) LECTURA: que el que "ve todas" pueda SELECT en todas las tablas de caja ── do $$ declare t text; tablas text[] := array[ 'caja_chica_config','caja_chica_rendiciones','caja_chica_movimientos', 'caja_chica_movimiento_items','caja_chica_adjuntos','caja_chica_acreditaciones', 'caja_chica_historial' ]; begin foreach t in array tablas loop if exists (select 1 from information_schema.tables where table_schema='public' and table_name=t) then execute format('drop policy if exists cc_ver_todas_select on public.%I', t); execute format('create policy cc_ver_todas_select on public.%I for select to authenticated using (public.cc_puede_ver_todas())', t); end if; end loop; end $$; -- ── 3) ESCRITURA: INSERT/UPDATE por permiso puntual (sin DELETE) ── do $$ declare rec record; begin for rec in select * from (values ('caja_chica_config', 'cc_configurar'), -- cambiar topes / alertas ('caja_chica_acreditaciones', 'cc_acreditar'), -- acreditar / reponer fondos ('caja_chica_movimientos', 'cc_aprobar'), -- aprobar / observar / excepción de comprobantes ('caja_chica_movimiento_items', 'cc_aprobar'), ('caja_chica_adjuntos', 'cc_aprobar'), -- adjuntos de aprobación ('caja_chica_rendiciones', 'cc_aprobar_rendicion') -- aprobar / cerrar rendiciones ) as v(tbl, perm) loop if exists (select 1 from information_schema.tables where table_schema='public' and table_name=rec.tbl) then execute format('drop policy if exists cc_gestion_insert on public.%I', rec.tbl); execute format('create policy cc_gestion_insert on public.%I for insert to authenticated with check (public.cc_perm(%L))', rec.tbl, rec.perm); execute format('drop policy if exists cc_gestion_update on public.%I', rec.tbl); execute format('create policy cc_gestion_update on public.%I for update to authenticated using (public.cc_perm(%L)) with check (public.cc_perm(%L))', rec.tbl, rec.perm, rec.perm); end if; end loop; -- Historial (log de auditoría): INSERT para cualquier gestor if exists (select 1 from information_schema.tables where table_schema='public' and table_name='caja_chica_historial') then execute 'drop policy if exists cc_gestion_insert on public.caja_chica_historial'; execute 'create policy cc_gestion_insert on public.caja_chica_historial for insert to authenticated with check (public.cc_es_gestor())'; end if; end $$; -- ── 4) Refrescar el cache del esquema de PostgREST ── notify pgrst, 'reload schema';
